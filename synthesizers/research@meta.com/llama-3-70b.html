About the Synthesizer

Access the Synthesizer

Python SDK

Use the SyftBox Python client to interact with the synthesizer:

#!/usr/bin/env python3
"""
Llama 3 70B synthesizer example
"""
import syft_hub as sh

def main():
    # Initialize the SyftBox client
    client = sh.get_client()
    
    # Connect to the Llama 3 70B synthesizer
    synthesizer = client.get_service(
        datasite="meta.com",
        name="llama-3-70b"
    )
    
    # Send a coding query
    response = synthesizer.chat(
        message="Write a Python function to implement a binary search algorithm with proper error handling.",
        temperature=0.7
    )
    
    # Display response
    print("Llama Response:")
    print("-" * 40)
    print(response.get('content', ''))
    print()
    print(f"Tokens used: {response.get('usage', {}).get('total_tokens', 'N/A')}")

if __name__ == "__main__":
    main()

HTTP API

Make direct HTTP requests to the synthesizer API:

# Chat API Example
curl -X POST "https://api.syftbox.net/v1/services/llama-3-70b/chat" \\
  -H "Content-Type: application/json" \\
  -H "Authorization: Bearer YOUR_API_KEY" \\
  -d '{
    "message": "Write a Python function to implement a binary search algorithm with proper error handling.",
    "temperature": 0.7,
    "max_tokens": 1000,
    "stream": false
  }'

      "command": "npx",
      "args": ["-y", "@syftbox/mcp-server"],
      "env": {
        "SYFTBOX_API_KEY": "YOUR_API_KEY",

    }
  }
}
